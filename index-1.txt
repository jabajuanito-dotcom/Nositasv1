      }

      inner += '<div class="meta"><span>' + formatDate(note.date) + '</span><span>·</span><span>' + typeLabel(note.type) + '</span></div>';
      inner += '</div>';

      if(currentTab === 'priority'){
        inner += '<button class="note-del" data-action="cycle" title="Cambiar prioridad"><svg viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2"><path d="M12 2l2.5 6.5L21 9l-5 4.5L17.5 21 12 17l-5.5 4L8 13.5 3 9l6.5-.5z"/></svg></button>';
      }

      card.innerHTML = inner;
      noteList.appendChild(card);
    });

    attachCardHandlers();
  }

  function attachCardHandlers(){
    document.querySelectorAll('.note-card').forEach(card=>{
      const id = card.dataset.id;

      if(currentTab === 'priority'){
        const btn = card.querySelector('[data-action="cycle"]');
        if(btn){
          btn.addEventListener('click', (e)=>{
            e.stopPropagation();
            const note = notes.find(n=>n.id===id);
            note.priority = cyclePriority(note.priority);
            save(); render();
          });
        }
      }

      if(currentTab === 'reorder'){
        card.addEventListener('click', ()=>{
          if(selectedForReorder === id){
            selectedForReorder = null;
          } else if(selectedForReorder === null){
            selectedForReorder = id;
          } else {
            const fromIdx = notes.findIndex(n=>n.id===selectedForReorder);
            const toIdx = notes.findIndex(n=>n.id===id);
            const moved = notes.splice(fromIdx,1)[0];
            notes.splice(toIdx,0,moved);
            selectedForReorder = null;
            save();
          }
          render();
        });
      }

      if(currentTab === 'delete'){
        attachDragAway(card, id);
      }
    });
  }

  function attachDragAway(card, id){
    let startX=0, startY=0, dx=0, dy=0, dragging=false;

    card.addEventListener('pointerdown', (e)=>{
      dragging = true;
      startX = e.clientX; startY = e.clientY;
      card.setPointerCapture(e.pointerId);
      card.style.transition = 'none';
      card.style.zIndex = 50;
    });
    card.addEventListener('pointermove', (e)=>{
      if(!dragging) return;
      dx = e.clientX - startX; dy = e.clientY - startY;
      card.style.transform = 'translate(' + dx + 'px, ' + dy + 'px) rotate(' + (dx*0.04) + 'deg)';
      const dist = Math.hypot(dx,dy);
      card.style.opacity = Math.max(0.15, 1 - dist/400);
    });
    function endDrag(){
      if(!dragging) return;
      dragging = false;
      card.style.transition = 'transform .25s ease, opacity .25s ease';
      const dist = Math.hypot(dx,dy);
      const offScreen = Math.abs(dx) > window.innerWidth*0.4 || Math.abs(dy) > 220 || dist > window.innerWidth*0.45;
      if(offScreen){
        card.style.transform = 'translate(' + (dx*3) + 'px, ' + (dy*3) + 'px) rotate(' + (dx*0.08) + 'deg)';
        card.style.opacity = '0';
        setTimeout(()=>{
          notes = notes.filter(n=>n.id!==id);
          save(); render();
        }, 220);
      } else {
        card.style.transform = 'translate(0,0)';
        card.style.opacity = '1';
      }
      dx=0; dy=0;
    }
    card.addEventListener('pointerup', endDrag);
    card.addEventListener('pointercancel', endDrag);
  }

  function renderCalendar(){
    const year = calDate.getFullYear(), month = calDate.getMonth();
    calLabel.textContent = calDate.toLocaleDateString('es-MX', {month:'long', year:'numeric'});
    calGrid.innerHTML = '';
    ['D','L','M','M','J','V','S'].forEach(d=>{
      const el = document.createElement('div'); el.className='dow'; el.textContent=d; calGrid.appendChild(el);
    });
    const firstDay = new Date(year, month, 1).getDay();
    const daysInMonth = new Date(year, month+1, 0).getDate();
    const today = new Date();
    const notesByDay = {};
    notes.forEach(n=>{
      const d = new Date(n.date);
      if(d.getFullYear()===year && d.getMonth()===month) notesByDay[d.getDate()] = true;
    });
    for(let i=0;i<firstDay;i++){
      const el = document.createElement('div'); el.className='cal-day empty'; calGrid.appendChild(el);
    }
    for(let day=1; day<=daysInMonth; day++){
      const el = document.createElement('div'); el.className='cal-day';
      el.textContent = day;
      if(today.getFullYear()===year && today.getMonth()===month && today.getDate()===day) el.classList.add('today');
      if(notesByDay[day]) el.classList.add('has-note');
      calGrid.appendChild(el);
    }
  }
